# Check whether queen placement is safe
def safe(b, r, c, n):

    for i in range(r):

        # Check same column and diagonal
        if b[i] == c or abs(b[i] - c) == abs(i - r):
            return False

    return True

# Backtracking function
def solve(r):

    # All queens placed
    if r == n:
        return True

    # Try placing queen in each column
    for c in range(n):

        if safe(board, r, c, n):

            board[r] = c

            # Recursive call
            if solve(r + 1):
                return True

    return False

# Input number of queens
n = int(input("Enter number of queens: "))

# Initialize board
board = [-1] * n

# Function call
if solve(0):

    print("\nSolution Found:\n")

    # Print chess board
    for i in range(n):

        for j in range(n):

            if board[i] == j:
                print("Q", end=" ")

            else:
                print(".", end=" ")

        print()

else:
    print("No Solution")
