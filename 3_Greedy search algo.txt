INF = 9999

# Input number of vertices
n = int(input("Enter number of vertices: "))

# Create adjacency matrix
graph = [[0 for i in range(n)] for j in range(n)]

# Input number of edges
e = int(input("Enter number of edges: "))

print("Enter edge (v1 v2 weight):")

# Input edges and weights
for i in range(e):

    v1, v2, w = map(int, input().split())

    graph[v1][v2] = w
    graph[v2][v1] = w

# Track visited vertices
visited = [False] * n

visited[0] = True

print("\nEdges in Minimum Spanning Tree:")

total_cost = 0

# Prim's Algorithm
for i in range(n - 1):

    minimum = INF

    x = 0
    y = 0

    for j in range(n):

        if visited[j]:

            for k in range(n):

                if (not visited[k]) and graph[j][k]:

                    if graph[j][k] < minimum:

                        minimum = graph[j][k]

                        x = j
                        y = k

    print(x, "-", y, "=", minimum)

    total_cost += minimum

    visited[y] = True

# Print minimum cost
print("Minimum Cost =", total_cost)
